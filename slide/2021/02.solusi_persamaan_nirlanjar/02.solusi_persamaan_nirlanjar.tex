\begin{frame}
	\frametitle{Rumusan Masalah}
	\begin{itemize}
		\item \textbf{Persoalan:} Temukan nilai $ x $ yang memenuhi persamaan
		\[f(x) = 0\]
		yaitu nilai $ x = s $ sedemikian sehingga $ f(s) = 0 $.
		\item Nilai $ x = s $ disebut \textbf{akar} persamaan $ f(x) = 0 $
	\end{itemize}
\end{frame}
